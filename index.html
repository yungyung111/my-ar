      /* 製作一個簡單的「掃描提示框」畫在螢幕中央，引導用家 */

        position: fixed;

    <!-- 掃描引導框，辨識成功後會用腳本隱藏 -->

      <div class="scanner-box"></div>

    </div>


    <!-- AR 舞台 (已更新識別檔路徑為 ./husktargets.mind) -->

    <a-scene mindar-image="imageTargetSrc: ./husktargets.mind; autoStart: true; uiLoading: no; uiError: no; uiScanning: no;" color-space="sRGB" renderer="colorManagement: true, physicallyCorrectLights" vr-mode-ui="enabled: false" device-orientation-permission-ui="enabled: false">

      

      <a-assets>

        <!-- 載入你的去背動畫影片 (已更新影片檔名為 ./husk-animation.webm) -->

        <video id="ar-video" src="./husk-animation.webm" loop="true" crossorigin="anonymous" playsinline webkit-playsinline muted></video>

      </assets>


      <a-camera position="0 0 0" look-controls="enabled: false"></a-camera>


      <!-- targetIndex: 0 代表你 husktargets.mind 裡的第一張識別圖 -->

      <a-entity id="ar-target" mindar-image-target="targetIndex: 0">

        <!-- 

          這裡放置你的 2D 動畫平面。

          width/height 代表寬高比例（可依影片比例微調，例如 16:9 可改為 width="1.6" height="0.9"）。

          position 的 Z 軸設為 0.05 讓動畫微微往前飄浮，更有立體感！

        -->

        <a-video src="#ar-video" width="1" height="1" position="0 0 0.05" rotation="0 0 0"></a-video>

      </a-entity>

    </a-scene>


    <script>

      const targetEl = document.querySelector('#ar-target');

      const video = document.querySelector('#ar-video');

      const overlay = document.querySelector('#scanning-overlay');


      // 當鏡頭「偵測到產品」時的動作

      targetEl.addEventListener("targetFound", event => {

        console.log("產品偵測成功！");

        overlay.style.display = "none"; // 隱藏掃描框

        video.play(); // 開始播放動畫

      });


      // 當鏡頭「離開產品」時的動作

      targetEl.addEventListener("targetLost", event => {

        console.log("產品離開視線");

        overlay.style.display = "flex"; // 重新顯示掃描框

        video.pause(); // 暫停動畫，節省手機效能

      });


      // 點擊螢幕可以解鎖聲音（手機瀏覽器安全機制限制自動播放帶聲音的影片）

      document.body.addEventListener("click", () => {

        video.muted = false;

      });

    </script>
